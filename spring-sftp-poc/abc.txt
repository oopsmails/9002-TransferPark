import com.jcraft.jsch.*;
import org.springframework.stereotype.Service;
import java.time.Instant;
import java.time.LocalDateTime;
import java.time.ZoneId;
import java.util.ArrayList;
import java.util.List;
import java.util.Vector;

@Service
public class SftpService {

    private final String HOST = "localhost";
    private final int PORT = 2222;
    private final String USER = "myuser";
    private final String PASSWORD = "mypassword";

    public List<FileDetail> listFileDetails(String directoryPath) throws JSchException, SftpException {
        List<FileDetail> details = new ArrayList<>();
        JSch jsch = new JSch();
        Session session = null;
        ChannelSftp channelSftp = null;

        try {
            session = jsch.getSession(USER, HOST, PORT);
            session.setPassword(PASSWORD);
            session.setConfig("StrictHostKeyChecking", "no");
            session.connect();

            channelSftp = (ChannelSftp) session.openChannel("sftp");
            channelSftp.connect();

            Vector<ChannelSftp.LsEntry> fileList = channelSftp.ls(directoryPath);
            
            for (ChannelSftp.LsEntry entry : fileList) {
                if (!entry.getAttrs().isDir()) {
                    SftpATTRS attrs = entry.getAttrs();
                    
                    details.add(new FileDetail(
                        entry.getFilename(),
                        attrs.getSize(),
                        convertToLocalDateTime(attrs.getMTime()), // Modified Time
                        convertToLocalDateTime(attrs.getATime()), // Access Time
                        attrs.getPermissionsString()
                    ));
                }
            }
        } finally {
            if (channelSftp != null) channelSftp.disconnect();
            if (session != null) session.disconnect();
        }
        return details;
    }

    /**
     * Converts JSch epoch seconds to LocalDateTime
     */
    private LocalDateTime convertToLocalDateTime(int epochSeconds) {
        return LocalDateTime.ofInstant(
            Instant.ofEpochSecond(epochSeconds), 
            ZoneId.systemDefault()
        );
    }
}
